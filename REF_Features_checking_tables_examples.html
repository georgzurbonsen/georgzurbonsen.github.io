table check (...) with 3 Parameters



In the example below, the resulting 'target' table contains 4 additional columns (in case they have not yet existed before).
The "Group Result" indicates "Not OK" because (at least) one row contains a mismatch.

Examples:

     include ( Support Library );

     table initialize( orig, {{ Name, Gender}, { Philippe, male }, { Pascale, emale }, { Michele, female }} );
     echo("Target table:");
     table list ( orig );

     table initialize( reference, { Gender, 'male,female' } );
     echo("Reference table:");
     table list ( reference );

     table copy table( orig, target );
     table check( target, reference, Gender );

     echo("Target table after checking:");
     table list ( target );

     table initialize( reference, { Gender, male, female } );
     echo("Reference table (male and female arranged vertically:");
     table list ( reference );

     table copy table( orig, target );
     table check( target, reference, Gender );

     echo("Target table after checking:");
     table list ( target );  //

Examples:
 include ( Support Library );

 table initialize( athletes,
 {
     { Name,     Gender, Hcp,    NTRP Lvl,   Sport   },
     { Mike,     male,   36, "",       Golf    },  // OK
     { Jan,      emale,  33, "",       Golf    },  // Gender wrong
     { Tina,     female, 36, "",       Golf    },  // Age wrong
     { Nico,     male,   99, "",       Golf    },  // HCP wrong
     { "",     male,   44, Lvl 1.0,    Golf    },  // Name missing
     { Lynn,     female, "",   Lvl 4.5,    Tennis  },  // OK
     { Yves,     female, "",   Lvl 2.0,    Squash  },  // Wrong sports
     { Zoë,     female, 18, Lvl 1.5,    Tennis  },  // OK
     { Chris,    male,   "",   Lvl 3.0,    Tennis  },  // OK
     { Willy,    male,   "",   Lvl 6.1,    Tennis  },  // Wrong level
     { Andy,     male,   "",   "",       Tennis  }   // Level missing

 } );    // OK again


 table copy table ( athletes, target );

 table keep columns( target, {Name, Sport, Hcp, NTRP Lvl} ); // Look how skilled they are
 table initialize( ref,
 {
     { Sport,    Hcp,        NTRP Lvl        },
     { Tennis,   '*',        'Lvl ?.0,Lvl ?.5'   },
     { Golf,     '0..54',    ''          }
 } );


 table check( target, ref, {Hcp, NTRP Lvl} );
 table list( target );


 table copy table ( athletes, target );
 table keep columns( target, {Name, Sport, Hcp, NTRP Lvl} ); // Look how skilled they are
 table check( target, ref, {Hcp, NTRP Lvl}, Sport );
 table list( target );
